180 void craft()
181 {
182 // Prepare
183 std::stringstream ss;
184 scaleFactorInverse = 1.0 / scaleFactor;
185 maxIntensity = minIntensity + deltaIntensity;
186 intensityCoefficient = 65535.0 / deltaIntensity;
187
188
189 // Build header
190 ss << "HELIOS++V" << HELIOS_VERSION;
191 std::string ssStr = ss.str();
192 size_t n = ssStr.length();
193 if(n > 31) n = 31;
194 memcpy(
195 lwHeader.generating_software,
196 ssStr.c_str(),
197 sizeof(char) * n
198 );
199 lwHeader.generating_software[n] = 0;
200 ss.str("");
201 lwHeader.version_major = U8(1);
202 lwHeader.version_minor = U8(0); // las version 1.0
203 time_t now = time(0);
204 tm *gmtm = gmtime(&now);
205 lwHeader.file_creation_day = gmtm->tm_yday;
206 lwHeader.file_creation_year = 1900 + gmtm->tm_year;
207 lwHeader.point_data_format = 1;
208 lwHeader.point_data_record_length = 28;
209 lwHeader.x_scale_factor = scaleFactor;
210 lwHeader.y_scale_factor = scaleFactor;
211 lwHeader.z_scale_factor = scaleFactor;
212 lwHeader.x_offset = F64(offset.x);
213 lwHeader.y_offset = F64(offset.y);
214 lwHeader.z_offset = F64(offset.z);
215 lwHeader.global_encoding = 0;
216 }

303 void _write(Measurement const &m, glm::dvec3 const & shift) override{
304 // Compute shifted position
305 glm::dvec3 shifted = m.position + shift - offset;
306
307 // Populate LAS point (base attributes)
308 lp.set_X(I32(shifted.x * scaleFactorInverse));
309 lp.set_Y(I32(shifted.y * scaleFactorInverse));
310 lp.set_Z(I32(shifted.z * scaleFactorInverse));
311 double intensityf = m.intensity;
312 if(intensityf < minIntensity) intensityf = minIntensity;
313 if(intensityf > maxIntensity) intensityf = maxIntensity;
314 U16 intensity = U16(
315 intensityCoefficient * (intensityf - minIntensity)
316 );
317 lp.set_intensity(intensity);
318
319 lp.set_return_number(U8(m.returnNumber));
320 lp.set_extended_return_number(U8(m.returnNumber));
321
322 lp.set_number_of_returns(U8(m.pulseReturnNumber));
323 lp.set_extended_number_of_returns(U8(m.pulseReturnNumber));
324
325 lp.set_classification(U8(m.classification) & CLASSIFICATION_MASK);
326 lp.set_extended_classification(U8(m.classification) & CLASSIFICATION_MASK);
327
328 lp.set_gps_time(F64(((double)m.gpsTime)/1000.0));
329
330 // Populate LAS point (extra bytes attributes)
331 lp.set_attribute(ewAttrStart, F64(m.echo_width));
332 lp.set_attribute(fwiAttrStart, I32(m.fullwaveIndex));
333 lp.set_attribute(hoiAttrStart, I32(std::stoi(m.hitObjectId)));
334 lp.set_attribute(ampAttrStart, F64(m.intensity));
335
336 // Write and add to inventory
337 lw->write_point(&lp);
338 lw->update_inventory(&lp);
339 }
