387  virtual size_t slopeFilter(VarType const tau){
388  // Obtain DX
389  arma::Mat<VarType> DX = arma::diff(this->getX());
390  DX.each_col() /= arma::diff(getTimeVector());
391  DX = arma::diff(DX);
392  // Filter by DX
393  size_t const m = DX.n_rows;
394  arma::Row<VarType> S(DX.n_cols, arma::fill::zeros);
395  vector<unsigned long long> remove;
396  for(size_t i = 0 ; i < m ; ++i){
397  S += DX.row(i);
398  VarType const maxDiff = arma::max(arma::abs(S));
399  if(maxDiff > tau) S = arma::zeros<arma::Row<VarType>>(DX.n_cols);
400  else remove.push_back(i+1);
401  }
402  dropRows(remove);
403  return remove.size();
404  }
